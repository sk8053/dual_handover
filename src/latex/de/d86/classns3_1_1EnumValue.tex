\subsection{Detailed Description}
Hold variables of type {\ttfamily enum} 

This class can be used to hold variables of any kind of enum.

This is often used with \hyperlink{classns3_1_1ObjectFactory}{Object\+Factory} and \hyperlink{namespacens3_1_1Config}{Config} to bind the value of a particular enum to an Attribute or \hyperlink{namespacens3_1_1Config}{Config} name. For example, 
\begin{DoxyCode}
Ptr<RateErrorModel> model = CreateObjectWithAttributes<RateErrorModel> (
  \textcolor{stringliteral}{"ErrorRate"}, DoubleValue (0.05),
  \textcolor{stringliteral}{"ErrorUnit"}, \hyperlink{classns3_1_1EnumValue_a1a57e1bf2fecec7119db9bfa0d1a28c7}{EnumValue} (\hyperlink{classns3_1_1RateErrorModel_a4da336ba2dfcb55e0b7d7a3e7e713c02aca366c7aebdc8be8b2ea287a7d3643a3}{RateErrorModel::ERROR\_UNIT\_PACKET}));

\hyperlink{group__config_ga2e7882df849d8ba4aaad31c934c40c06}{Config::SetDefault} (\textcolor{stringliteral}{"ns3::RipNg::SplitHorizon"},
                    \hyperlink{classns3_1_1EnumValue_a1a57e1bf2fecec7119db9bfa0d1a28c7}{EnumValue} (\hyperlink{classns3_1_1RipNg_a2baa3148870b411bccfe688079fb40a7a87060df7c85bafc99b78440f423a171b}{RipNg::NO\_SPLIT\_HORIZON}));
\end{DoxyCode}
